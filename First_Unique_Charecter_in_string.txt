Given a string s, find the first non-repeating character in it and return its index.
If it does not exist, return -1.

solution in python3 =>

class Solution:
    def firstUniqChar(self, s: str) -> int:
        h={}
        for c in s:
            h[c]=h.get(c,0)+1
        for i,c in enumerate(s):
            if h[c]==1:
                return i
