Given an integer array nums, return true if any value appears at 
least twice in the array, and return false if every element is distinct.

solution2 in python3 =>

class Solution:
    def containsDuplicate(self, nums: List[int]) -> bool:
        hset=set()
        for i in nums:
            if i in hset:
                return True
            else:
                hset.add(i)
        return False
